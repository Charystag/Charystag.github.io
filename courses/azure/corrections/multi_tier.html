1. Ajout de Données Dynamiques

Ajoutez la route /add_user dans db_server.py :

@app.route('/add_user/<name>', methods=['POST'])
def add_user(name):
    conn = sqlite3.connect('/home/azureuser/db/demo.db')
    cursor = conn.cursor()
    cursor.execute("INSERT INTO users (name) VALUES (?)", (name,))
    conn.commit()
    conn.close()
    return jsonify({"message": f"Utilisateur {name} ajouté avec succès."})
